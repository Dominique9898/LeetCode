/*
两个指针最多
如果相交，会提前相遇在相交节点。此时返回相交节点。
如果不相交，则各自走过headA链表长度 + headB链表长度的距离，指向null。此时返回null.
*/

var getIntersectionNode = function(headA, headB) {
    var p1 = headA, p2 = headB;
    while (p1 != p2) {
        p1 = p1 ? p1.next : headB;
        p2 = p2 ? p2.next : headA;
    }
    return p1;
};
